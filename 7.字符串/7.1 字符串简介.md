# 基本介绍
字符串以数组形式存在，以数组或指针形式访问
字符串不能进行运算
字符串以字符数组形式存在

# 语法
字符串以'\0'或0结尾，其中0标志字符串结束，并不作为字符串的一部分
    char word[] = {'H', 'e', 'l', 'l', 'o', '\0' }; 
    char word[] = {'H', 'e', 'l', 'l', 'o', 0 }; 
    char word[] = "Hello";
        等价于上面两行
        char word[]是字符数组，存储于栈 (Stack)，	可读、可写
        栈是内存管理中的一个物理区域，后进先出
    char *p = "Hello";
        字符串常量，不能修改,指向代码段
        char *p是字符串指针，属于代码/数据段，仅可读
        指针a本身在栈上，但它指向的字节（malloc的区域）在堆上。

    printf("p = %p\n", p);
    printf("word = %p\n", word); 

//构造字符串，用数组构造；处理字符串，用指针构造

scanf("%s", word); //读入单词，到空格或换行结束
scanf("%7s", word); //读入最多7个字符，最后一个位置给\0
printf("%s\n", word); //打印字符串，遇到\0结束

char a[][10]

const int *p(const在*前)：指针所的东西不能修改
int *const p（const在*后）：指针本身不能修改
