# 既声明结构类型point，又定义变量（p1、p2）
  - struct point{
        int x;
        int y;          
    };
    struct point p1, p2;   

  - struct point{
        int x;
        int y;
    } p1, p2;            

# 只定义变量（p1和p2）
    struct{
        int x;
        int y;
    } p1, p2;            

# 结构赋值
    p1 = (struct point){5, 10};
    p1 = p2; //结构变量可以相互赋值,数组变量不可以

# 结构指针
    必须用&取地址: struct date *pDate = &today; 

    如：
    struct date {
        int month;
        int day;
        int year;
    }myday;

    struct date *p = &myday;

    以下两种形式等价（将12赋给month）：
    (*p).month = 12;
    p->month = 12;

# 结构可以作为函数参数（类似int）
    int numberofdays(struct date d);
